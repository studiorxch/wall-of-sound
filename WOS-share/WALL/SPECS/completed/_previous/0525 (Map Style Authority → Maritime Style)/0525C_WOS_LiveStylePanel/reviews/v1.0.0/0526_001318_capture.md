# 1. CORE ARCHITECTURAL DOCTRINE

The `LiveStylePanel` provides a developer-focused, live interaction dashboard for testing and modifying presentation configurations in real-time. It enables swift, data-driven tuning of the world aesthetics without altering source files or hardcoding values into structural render layers.

- **Presentation systems interpret the world. They do NOT create runtime truth.**
- **Live tooling may tune appearance; it may not mutate reality.**
- **Developer iteration is not runtime authority.**
- **One active override authority at a time.**

import SBE from "./maritimeStyleRegistry.js";

export class LiveStylePanelModule {
  private activeMode: StylePanelMode = "INACTIVE";
  private currentDraft: StyleOverrideDraft | null = null;
  private currentValidationResult: StylePanelValidationResult = { pass: true, errors: [] };
  private globalStyleAuthority: any;

  constructor(injectedStyleAuthority: any) {
    this.globalStyleAuthority = injectedStyleAuthority;
  }

  public createStyleOverrideDraft(targetDomain: StylePanelTargetDomain, targetLayerOrClass: string): StyleOverrideDraft {
    this.currentDraft = {
      draftId: `style-draft::${Date.now()}`,
      targetDomain,
      targetLayer: targetDomain === "MAP" ? targetLayerOrClass as MapStyleLayerKey : undefined,
      maritimeClassKey: targetDomain === "MARITIME" ? targetLayerOrClass : undefined,
      values: {},
      createdAtMs: Date.now(),
      updatedAtMs: Date.now(),
    };
    this.activeMode = "DRAFT";
    return this.currentDraft;
  }

  public updateDraftField(fieldKey: string, value: unknown): void {
    if (!this.currentDraft) return;
    this.currentDraft.values[fieldKey] = value;
    this.currentDraft = { ...this.currentDraft, updatedAtMs: Date.now() };
    this.validateStyleOverrideDraft();
  }

  public validateStyleOverrideDraft(): StylePanelValidationResult {
    const errors: StylePanelValidationError[] = [];
    if (!this.currentDraft) {
      return { pass: true, errors: [] };
    }

    const draft = this.currentDraft;
    const allowedFields = [
      "baseColor", "shimmerStrength", "reflectionOpacity", "currentBandAlpha", "coastlineContrast", "harborDarkness",
      "landColorHex", "districtContrast", "coastlineVisibility", "infrastructureShadowStrength", "nighttimeDarkness",
      "arterialOpacity", "localRoadOpacity", "glowStrength", "labelSuppression", "nighttimeFade",
      "density", "opacity", "suppressionStrength", "fogAlpha", "hazeStrength", "grainOpacity", "glowRadius", "bloomSoftness",
      "hudOpacity", "scannerStrength", "typographyGlow", "telemetrySoftness", "noiseSuppression",
      "hullColorHex", "deckColorHex", "accentColorHex", "strokeWidthPx", "compactScaleMultiplier", "detailedScaleMultiplier",
      "silhouetteWeight", "markerRadiusPx", "farLightAlpha", "farLightHaloPx", "twinkleStrength", "twinkleRateHz",
      "lowVisibilityDamping", "classTintStrength", "visualAlphaMultiplier", "edgeSoftnessScalar",
      "densitySuppressionStrength", "headingVisualSmoothing", "visualEasingMs", "backgroundAlpha", "borderAlpha",
      "borderRadiusPx", "classAccentStrength", "glowStrength", "fadeInMs", "holdMs", "fadeOutMs", "maxWidthPx",
      "clutterSuppressionStrength", "farLightSuppressionStrength", "wakeSuppressionStrength", "labelVisualSuppressionStrength"
    ];

    for (const fieldKey of Object.keys(draft.values)) {
      if (!allowedFields.includes(fieldKey)) {
        errors.push({
          fieldKey,
          message: "Field is not owned by LiveStylePanel presentation authority.",
          authorityViolation: true,
        });
      }

      if (fieldKey === "twinkleRateHz") {
        const valueNum = Number(draft.values[fieldKey]);
        if (valueNum > MAX_TWINKLE_RATE_HZ) {
          errors.push({
            fieldKey,
            message: `twinkleRateHz may not exceed ${MAX_TWINKLE_RATE_HZ} Hz.`,
            authorityViolation: true,
          });
        }
      }

      if (fieldKey === "holdMs") {
        const valueNum = Number(draft.values[fieldKey]);
        if (valueNum > MAX_HOVER_HOLD_MS) {
          errors.push({
            fieldKey,
            message: `hover hold may not exceed ${MAX_HOVER_HOLD_MS}ms.`,
            authorityViolation: true,
          });
        }
      }

      // Enforcement rule: clamp standard alpha properties to safe limits
      if (fieldKey.endsWith("Alpha") || fieldKey.endsWith("Opacity") || fieldKey.endsWith("Strength")) {
        const valueNum = Number(draft.values[fieldKey]);
        if (valueNum < MIN_ALPHA_VALUE || valueNum > MAX_ALPHA_VALUE) {
          errors.push({
            fieldKey,
            message: `Field properties must be bound between ${MIN_ALPHA_VALUE} and ${MAX_ALPHA_VALUE}.`,
            authorityViolation: true,
          });
        }
      }
    }

    this.currentValidationResult = { pass: errors.length === 0, errors };
    if (!this.currentValidationResult.pass) {
      this.activeMode = "INVALID";
    } else {
      this.activeMode = "DRAFT";
    }
    return this.currentValidationResult;
  }

  public applyDraftAsLiveOverride(): StylePanelValidationResult {
    this.validateStyleOverrideDraft();
    if (!this.currentValidationResult.pass || !this.currentDraft) {
      return this.currentValidationResult;
    }

    // Single writer execution pass: transform draft items safely into the map authority
    const consolidatedOverride = { ...this.currentDraft.values };
    this.globalStyleAuthority.setSingleLiveOverride(consolidatedOverride);
    this.activeMode = "LIVE_OVERRIDE";

    return this.currentValidationResult;
  }

  public clearLiveStyleOverride(): void {
    this.globalStyleAuthority.clearLiveOverride();
    this.currentDraft = null;
    this.currentValidationResult = { pass: true, errors: [] };
    this.activeMode = "INACTIVE";
  }

  public getSnapshot(activeManifestId: string | null): StylePanelSnapshot {
    return {
      mode: this.activeMode,
      activeOverrideId: this.activeMode === "LIVE_OVERRIDE" ? "live-active-override" : null,
      targetDomain: this.currentDraft ? this.currentDraft.targetDomain : null,
      draft: this.currentDraft,
      validation: this.currentValidationResult,
      manifestId: activeManifestId
    };
  }
}
