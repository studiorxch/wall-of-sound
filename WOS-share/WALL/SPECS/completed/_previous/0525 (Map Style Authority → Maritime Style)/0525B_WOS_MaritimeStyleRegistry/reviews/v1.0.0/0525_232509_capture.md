# 1. CORE ARCHITECTURAL DOCTRINE

This specification defines the authoritative maritime symbolic style registry for WOS harbor and vessel rendering. It extracts concrete maritime visual grammar away from global layout authorities into a separate presentation domain plugin.

- **Presentation systems interpret the world. They do NOT create runtime truth.**
- **2D owns truth. 2.5D owns presentation.**
- **Maritime style expresses continuity; it does not create continuity.**
- **Far vessels are atmospheric harbor infrastructure.**

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Vessel-class visual profiles, color palettes, and structural scale multipliers.
- Far-light twinkle factors, halo dimensions, and ambient visibility dampening.
- Wake visual alpha treatments, edge softness factors, and density suppression scaling.
- Hover-card presentation traits, border radiuses, opacities, and visual transition times.
- Visual density-response mitigation lookups and taxonomic fallback routing paths.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Telemetry positions, velocity configurations, headings, or status fields within `AISRuntime`.
- Active segment coordinates, decay cycles, or trailing buffers inside `MaritimeWakeAuthority`.
- Priority tracks, targeting queues, or tier determinations inside `MaritimePopulationHierarchy`.
- The structural visibility assignments (`FULL`, `REDUCED`, `SILHOUETTE`, `MARKER_ONLY`, `LIGHT_ONLY`, `ATMOSPHERIC_HIDDEN`) calculated by `AtmosphericReadability`.

export class MaritimeStyleRegistryModule {
  private registry: MaritimeStyleRegistry;

  constructor() {
    this.registry = { ...maritimeStyleRegistry };
  }

  public normalizeVesselClass(rawClass: string | null | undefined): MaritimeVesselStyleKey {
    if (!rawClass) return "default";
    const normalized = rawClass.trim().toLowerCase();
    if (normalized in this.registry) {
      return normalized as MaritimeVesselStyleKey;
    }
    return "default";
  }

  public resolveVesselStyle(rawClass: string | null | undefined): ClassStyleResolutionResult {
    const resolvedClass = this.normalizeVesselClass(rawClass);
    const usedFallback = resolvedClass === "default" && rawClass !== "default";
    return {
      requestedClass: rawClass ?? null,
      resolvedClass,
      usedFallback,
      vesselStyle: JSON.parse(JSON.stringify(this.registry[resolvedClass])),
    };
  }

  public applyVisibilityClassToStyle(vesselStyle: VesselStyle, visibilityClass: VisibilityClass): VesselStyle {
    const cloned = JSON.parse(JSON.stringify(vesselStyle)) as VesselStyle;
    switch (visibilityClass) {
      case "FULL":
        return cloned;
      case "REDUCED":
        return {
          ...cloned,
          symbolic: { ...cloned.symbolic, strokeWidthPx: cloned.symbolic.strokeWidthPx * 0.85 },
          wakePresentation: { ...cloned.wakePresentation, visualAlphaMultiplier: cloned.wakePresentation.visualAlphaMultiplier * 0.65 }
        };
      case "SILHOUETTE":
        return {
          ...cloned,
          lighting: { ...cloned.lighting, farLightAlpha: cloned.lighting.farLightAlpha * 0.75 },
          wakePresentation: { ...cloned.wakePresentation, visualAlphaMultiplier: 0.0 }
        };
      case "MARKER_ONLY":
        return {
          ...cloned,
          symbolic: { ...cloned.symbolic, detailedScaleMultiplier: 0.0 },
          wakePresentation: { ...cloned.wakePresentation, visualAlphaMultiplier: 0.0 }
        };
      case "LIGHT_ONLY":
        return {
          ...cloned,
          symbolic: { ...cloned.symbolic, compactScaleMultiplier: 0.0, detailedScaleMultiplier: 0.0 },
          wakePresentation: { ...cloned.wakePresentation, visualAlphaMultiplier: 0.0 }
        };
      case "ATMOSPHERIC_HIDDEN":
        return {
          ...cloned,
          symbolic: { ...cloned.symbolic, compactScaleMultiplier: 0.0, detailedScaleMultiplier: 0.0, markerRadiusPx: 0.0 },
          lighting: { ...cloned.lighting, farLightAlpha: 0.0, farLightHaloPx: 0.0, twinkleStrength: 0.0 },
          wakePresentation: { ...cloned.wakePresentation, visualAlphaMultiplier: 0.0 }
        };
      default:
        return cloned;
    }
  }

  public applyDensityPressureToStyle(vesselStyle: VesselStyle, clutterPressure: number): VesselStyle {
    const pressure = Math.max(0, Math.min(1, clutterPressure));
    const cloned = JSON.parse(JSON.stringify(vesselStyle)) as VesselStyle;
    const density = cloned.densityResponse;

    return {
      ...cloned,
      lighting: {
        ...cloned.lighting,
        farLightAlpha: cloned.lighting.farLightAlpha * (1 - pressure * density.farLightSuppressionStrength),
      },
      wakePresentation: {
        ...cloned.wakePresentation,
        visualAlphaMultiplier: cloned.wakePresentation.visualAlphaMultiplier * (1 - pressure * density.wakeSuppressionStrength),
      },
      hoverCardPresentation: {
        ...cloned.hoverCardPresentation,
        glowStrength: cloned.hoverCardPresentation.glowStrength * (1 - pressure * density.clutterSuppressionStrength),
      },
    };
  }
}
