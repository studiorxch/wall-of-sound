# 1. CORE ARCHITECTURAL DOCTRINE

The `VisibilityClassRuntime` acts as a deterministic interpretive bridge, converting environmental atmospheric conditions, zoom layers, crowd density levels, and visual style presets into explicit presentation parameters (`visibilityClass`).

- **Visibility classes constrain presentation. They do NOT create runtime truth.**
- **2D owns truth. 2.5D owns presentation.**
- **Hidden vessels may not be visually elevated by style systems.**
- **Presentation systems may tighten visibility; they may not fabricate observability.**

---

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- The systemic implementation of visibility resolution policies and matching logic rings.
- Frame-by-frame structural parsing of incoming environmental telemetry into static constraint tokens.
- Strict evaluation and tracking of token degradation along the Downward-Only priority index.
- Generation of detailed diagnostic snapshots tracking active visibility class allocations.
- Real-time resolution pipelines processing singular items or batched data streams.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Tracker coordinates, heading entries, velocity values, or metadata fields within `AISRuntime`.
- Active trail segments, point maps, or lifespans governed by `MaritimeWakeAuthority`.
- Priority assignments or triage bucket categorizations within `MaritimePopulationHierarchy`.
- Base atmospheric tracking metrics assigned by `AtmosphericReadability`.
- Layout parameters, interface configurations, or theme lists inside `SurfaceStylePresetRuntime`.
- Global style registries, active drawing sheets, or text tracking states inside `MarineRenderer`.

EXECUTABLE RESOLUTION ENGINETypeScriptexport class VisibilityClassRuntimeModule {
  private resolvedCounter: number = 0;
  private classAllocationMap: Record<VisibilityClass, number>;
  private activePresetId: string | null = null;

  constructor() {
    this.classAllocationMap = this.createEmptyCounter();
  }

  private createEmptyCounter(): Record<VisibilityClass, number> {
    return {
      FULL: 0,
      REDUCED: 0,
      SILHOUETTE: 0,
      MARKER_ONLY: 0,
      LIGHT_ONLY: 0,
      ATMOSPHERIC_HIDDEN: 0,
    };
  }

  private visibilityRank(value: VisibilityClass): number {
    return VISIBILITY_ORDER.indexOf(value);
  }

  public tightenVisibility(base: VisibilityClass, candidate: VisibilityClass): VisibilityClass {
    return this.visibilityRank(candidate) > this.visibilityRank(base) ? candidate : base;
  }

  public resolveVisibilityClass(input: VisibilityResolutionInput): VisibilityResolutionResult {
    let resolved = input.upstreamVisibilityClass || DEFAULT_VISIBILITY_CLASS;
    let sourceAuthority: VisibilityResolutionResult["sourceAuthority"] = input.upstreamVisibilityClass
      ? "ATMOSPHERIC_READABILITY"
      : "FALLBACK";

    const original = resolved;

    // Evaluate structural clutter pressures
    if (input.clutterPressure >= HIGH_DENSITY_THRESHOLD) {
      const NextTighten = this.tightenVisibility(resolved, "MARKER_ONLY");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "DENSITY";
      }
    } else if (input.clutterPressure >= MEDIUM_DENSITY_THRESHOLD) {
      const NextTighten = this.tightenVisibility(resolved, "REDUCED");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "DENSITY";
      }
    }

    // Evaluate structural camera distances and zoom levels
    if (input.distanceKm !== null && input.distanceKm >= FAR_DISTANCE_KM) {
      const NextTighten = this.tightenVisibility(resolved, "LIGHT_ONLY");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "ZOOM";
      }
    } else if (input.zoom !== null && input.zoom <= LOW_ZOOM_THRESHOLD) {
      const NextTighten = this.tightenVisibility(resolved, "MARKER_ONLY");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "ZOOM";
      }
    }

    // Evaluate active structural thematic style overrides
    if (input.activePresetCategory === "SIGNAL_DRIFT") {
      const NextTighten = this.tightenVisibility(resolved, "SILHOUETTE");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "PRESET";
      }
    } else if (input.activePresetCategory === "BROADCAST_FAILURE") {
      const NextTighten = this.tightenVisibility(resolved, "LIGHT_ONLY");
      if (NextTighten !== resolved) {
        resolved = NextTighten;
        sourceAuthority = "PRESET";
      }
    }

    // Enforce isolation protocols on ghost targets
    if (input.populationTier === "GHOST") {
      resolved = this.tightenVisibility(resolved, "LIGHT_ONLY");
    }

    // Update internal frame metrics counters
    this.resolvedCounter++;
    this.classAllocationMap[resolved]++;
    if (input.activePresetId) {
      this.activePresetId = input.activePresetId;
    }

    return {
      entityId: input.entityId,
      visibilityClass: resolved,
      sourceAuthority,
      tightenedFrom: original === resolved ? null : original,
      factors: {
        clutterPressure: input.clutterPressure,
        zoom: input.zoom,
        distanceKm: input.distanceKm,
        atmosphereStrength: input.atmosphereStrength,
        readabilityBias: input.readabilityBias,
        densityBias: input.densityBias,
      },
    };
  }

  public resolveBatch(inputs: readonly VisibilityResolutionInput[]): readonly VisibilityResolutionResult[] {
    this.resolvedCounter = 0;
    this.classAllocationMap = this.createEmptyCounter();
    
    const output: VisibilityResolutionResult[] = [];
    for (let i = 0; i < inputs.length; i++) {
      output.push(this.resolveVisibilityClass(inputs[i]));
    }
    return output;
  }

  public getSnapshot(): VisibilityClassRuntimeSnapshot {
    return {
      version: "1.0.0",
      activePresetId: this.activePresetId,
      resolvedCount: this.resolvedCounter,
      classCounts: { ...this.classAllocationMap },
      lastResolvedAtMs: Date.now(),
    };
  }
}
