/**
 * Gets the effective profile for a vessel, applying zoom-based scaling.
 */
function getEffectiveProfile(vessel, currentZoom) {
    let baseProfile = WAKE_PROFILES[vessel.classKey] || WAKE_PROFILES.cargo;
    
    // Zoom-based LOD suppression
    let zoomFactor = 0;
    if (currentZoom <= WAKE_CONFIG.DEFAULT_MIN_ZOOM) {
        zoomFactor = 0; // Far zoom: no wake
    } else if (currentZoom >= WAKE_CONFIG.DEFAULT_FULL_ZOOM) {
        zoomFactor = 1; // Full wake
    } else {
        // Linear interpolation between min and full zoom
        zoomFactor = (currentZoom - WAKE_CONFIG.DEFAULT_MIN_ZOOM) / 
                     (WAKE_CONFIG.DEFAULT_FULL_ZOOM - WAKE_CONFIG.DEFAULT_MIN_ZOOM);
        // Apply an ease-out curve for a more natural fade-in
        zoomFactor = Math.pow(zoomFactor, 1.5);
    }
    
    if (zoomFactor <= 0) return null;
    
    // Create an instance of the profile with zoom-applied scales
    return {
        ...baseProfile,
        effectiveLength: baseProfile.lengthScale * (0.5 + zoomFactor * 0.5), // Wake grows with zoom
        effectiveAlpha: baseProfile.maxWakeAlpha * zoomFactor,
        effectiveGlow: baseProfile.glowStrength * zoomFactor,
        zoomFactor: zoomFactor,
    };
}

/**
 * Draws a gradient-based wake line with soft caps.
 */
function drawSoftWakeLine(ctx, points, profile, vesselHeading) {
    if (!points || points.length < 2) return;
    
    // Apply class-specific alpha and zoom factor
    const globalAlpha = profile.effectiveAlpha * profile.alphaScale;
    if (globalAlpha <= 0.01) return;
    
    ctx.save();
    ctx.globalAlpha = globalAlpha;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';
    ctx.lineWidth = Math.min(WAKE_CONFIG.MAX_LINE_WIDTH_PX, 
                             WAKE_CONFIG.MIN_LINE_WIDTH_PX + profile.widthScale * 4);
    
    // Create a gradient along the path length
    const gradient = ctx.createLinearGradient(points[0].x, points[0].y, 
                                               points[points.length-1].x, points[points.length-1].y);
    
    // Use the standardized gradient stops
    for (const stop of WAKE_CONFIG.GRADIENT_STOPS) {
        gradient.addColorStop(stop.pos, `rgba(255, 255, 255, ${stop.alpha * profile.alphaScale})`);
    }
    
    ctx.strokeStyle = gradient;
    
    // Optional: subtle glow for close zoom
    if (profile.effectiveGlow > 0 && profile.zoomFactor > 0.7) {
        ctx.shadowBlur = 6;
        ctx.shadowColor = `rgba(255, 255, 240, ${profile.effectiveGlow * 0.5})`;
    }
    
    // Draw the main wake line
    ctx.beginPath();
    ctx.moveTo(points[0].x, points[0].y);

/**
 * Draws turbulence filaments for TURBULENT mode.
 */
function drawTurbulence(ctx, vesselPos, vesselHeading, profile, vesselSpeed) {
    if (profile.turbulenceCount === 0) return;
    
    const count = Math.min(profile.turbulenceCount, WAKE_CONFIG.MAX_TURBULENCE_FILAMENTS);
    const speedFactor = Math.min(1, vesselSpeed / 12); // 12 knots = full
    
    ctx.save();
    ctx.globalAlpha = profile.effectiveAlpha * 0.6 * speedFactor;
    ctx.lineCap = 'round';
    ctx.lineWidth = 2;
    
    for (let i = 0; i < count; i++) {
        // Use deterministic jitter based on vessel ID to avoid repetition
        const seed = (vesselPos.x * 1000 + vesselPos.y * 100 + i) % 1000 / 1000;
        const angleOffset = (Math.sin(seed * Math.PI * 2) * 0.6) - 0.3; // -0.3 to 0.3 rad offset
        const lengthOffset = 0.5 + Math.cos(seed * Math.PI * 1.3) * 0.3;
        
        const angle = vesselHeading + angleOffset;
        const length = (profile.lengthScale * 8 * lengthOffset) * speedFactor;
        
        const endX = vesselPos.x + Math.sin(angle) * length;
        const endY = vesselPos.y - Math.cos(angle) * length;
        
        ctx.beginPath();
        ctx.moveTo(vesselPos.x, vesselPos.y);
        ctx.lineTo(endX, endY);
        
        // Turbulence lines have a secondary gradient
        const grad = ctx.createLinearGradient(vesselPos.x, vesselPos.y, endX, endY);
        grad.addColorStop(0, 'rgba(220, 240, 255, 0.8)');
        grad.addColorStop(1, 'rgba(200, 220, 240, 0)');
        ctx.strokeStyle = grad;
        ctx.stroke();
    }
    ctx.restore();
}

// OPTIONAL - Load only when _wos.debug === true or URL param ?wakedebug=1

(function() {

        console.error("[WakeDebug] Core _wos.wakeSignature API not found. Cannot attach debugger.");

            <div id="wake-debug-data">---</div>
            <button id="wake-debug-close" style="margin-top:8px;">Close</button>
        `;
        document.body.appendChild(debugOverlay);
        
        document.getElementById('wake-debug-close').onclick = () => {
            debugOverlay.remove();
            debugOverlay = null;
            if (selectedVessel) selectedVessel = null;
        };
    }
    
    // Hook into the map's click handler (this is a simplification)
    function onVesselClick(vessel) {
        if (!debugOverlay) createDebugPanel();
        selectedVessel = vessel;
        
        const profile = _wos.wakeSignature.profile(vessel.classKey);
        const dataDiv = document.getElementById('wake-debug-data');
        if (profile) {
            dataDiv.innerHTML = `
                <hr><b>${vessel.classKey}</b> | Speed: ${vessel.speed} kts<br>
                Mode: ${profile.mode}<br>
                Length: ${profile.lengthScale}<br>
                Alpha: ${profile.maxWakeAlpha}<br>
                Turbulence: ${profile.turbulenceCount}<br>
                <span style="color:#aaa;">click map to refresh</span>
            `;
        } else {
            dataDiv.innerHTML = `<hr>No profile for ${vessel.classKey}`;
        }
    }
    
    // Expose the click handler to the main map system
    // This assumes your map has a 'vessel:click' event or similar
    if (window.map && window.map.on) {
        window.map.on('vessel:click', onVesselClick);
        console.log("[WakeDebug] Attached to map vessel clicks.");
    } else {
        console.warn("[WakeDebug] Could not attach to map events. Manual trigger only.");
        window._wos.wakeSignature.inspectVessel = onVesselClick;
    }
    
    console.log("[WakeDebug] Active. Use _wos.wakeSignature.inspectVessel(vesselObject) or click a vessel.");
