# 1. CORE ARCHITECTURAL DOCTRINE

The `ActiveWakePolish` architecture defines the drawing and visual smoothing constraints for live, entity-attached vessel wake rendering pipelines.

- **Active wakes communicate live motion. They do not simulate water.**
- **Stateless Visual Processing:** Wakes must be generated on the fly directly from the vessel's current velocity, heading, and class matrix, without using historical data frames.
- **2D owns truth. 2.5D owns presentation.**
- **Strict Visual Softness Limits:** Sharp geometric borders, blocky tiles, or screen-space tracking fields are completely prohibited. All paths use smooth color gradients and rounded path joins.
- **Water Memory Isolation:** Active wake rendering must remain completely decoupled from historical tracking arrays. Wakes must render cleanly when `showMaritimeWaterMemory` is explicitly turned off.

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Configuration profiles containing length, opacity, scale, and softness rules for all 11 canonical classes.
- Linear and split-V gradient drawing loops for active vessel footprints.
- Taper configurations used to blend line transitions cleanly into ambient water textures.
- Zoom-dependent level-of-detail visibility tracking for vector trails.
- Localized control-point coordinates used to inject deterministic variations into active paths.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Active coordinates, headings, velocity vectors, or tracking attributes inside `AISRuntime`.
- Underlying state configurations or lifetime metrics belonging to `MaritimeWakeAuthority`.
- Global visibility ratings or environmental density fields evaluated by `VisibilityClassRuntime`.
- Active system configuration switches (`SBE.runtimeFlags`).
- Global canvas dimensions, layer order definitions, or transform contexts managed by the renderer.

6. EXECUTABLE STATELESS RE-RENDERING ENGINETypeScriptexport class ActiveWakePolishModule {
  private profiles: Map<string, ActiveWakePolishProfile> = new Map();
  private debugMode: boolean = WAKE_DEBUG_DEFAULT;

  constructor() {
    this.initializeProfiles();
    this.ENFORCE_WATER_MEMORY_CONTAINMENT();
  }

  private ENFORCE_WATER_MEMORY_CONTAINMENT(): void {
    // Explicit global architecture isolation guardrail
    if (typeof globalThis !== 'undefined') {
      (globalThis as any).SBE = (globalThis as any).SBE || {};
      (globalThis as any).SBE.runtimeFlags = (globalThis as any).SBE.runtimeFlags || {};
      (globalThis as any).SBE.runtimeFlags.showMaritimeWaterMemory = false;
    }
  }

  private initializeProfiles(): void {
    for (const [key, profile] of Object.entries(WAKE_POLISH_PROFILES)) {
      this.profiles.set(key, profile);
    }
  }

  public getProfile(classKey: string): ActiveWakePolishProfile {
    return this.profiles.get(classKey?.toLowerCase()) || this.profiles.get("default")!;
  }

  public setDebug(state: boolean): void {
    this.debugMode = state;
  }

  public calculateZoomAlphaScale(zoom: number, profile: ActiveWakePolishProfile): number {
    if (zoom <= profile.minVisibleZoom) return 0.0;
    if (zoom >= profile.fullVisibleZoom) return 1.0;
    return (zoom - profile.minVisibleZoom) / (profile.fullVisibleZoom - profile.minVisibleZoom);
  }

  public renderActiveWake(ctx: CanvasRenderingContext2D, input: ActiveWakeRenderInput): void {
    if (input.visibilityClass === "ATMOSPHERIC_HIDDEN" || input.visibilityClass === "LIGHT_ONLY") return;
    if (input.speedKts < 0.5) return;

    const profile = this.getProfile(input.classKey);
    if (input.currentZoom < profile.minVisibleZoom) return;

    const zoomScale = this.calculateZoomAlphaScale(input.currentZoom, profile);
    const targetAlpha = profile.maxWakeAlpha * profile.alphaScale * zoomScale;
    
    ctx.save();
    
    // Position alignment and spatial translation transformations
    ctx.translate(input.x, input.y);
    ctx.rotate((input.headingDeg * Math.PI) / 180);

    // Apply strict softness rules across lines and geometric caps
    ctx.lineCap = "round";
    ctx.lineJoin = "round";

    const baseLength = input.speedKts * profile.lengthScale * 4.0 * input.topologyScale;
    const baseWidth = Math.max(
      MIN_WAKE_LINE_WIDTH_PX,
      Math.min(MAX_WAKE_LINE_WIDTH_PX, input.speedKts * profile.widthScale * 0.4 * input.topologyScale)
    );

    switch (profile.mode) {
      case "LINEAR":
        this.drawLinearWake(ctx, baseLength, baseWidth, targetAlpha, profile);
        break;
      case "SPLIT_V":
        this.drawSplitVWake(ctx, baseLength, baseWidth, targetAlpha, profile);
        break;
      case "TURBULENT":
        this.drawTurbulentWake(ctx, baseLength, baseWidth, targetAlpha, input.vesselId, profile);
        break;
      case "DRIFT":
        this.drawDriftWake(ctx, baseLength, baseWidth, targetAlpha, profile);
        break;
      case "DISCIPLINED":
        this.drawLinearWake(ctx, baseLength * 0.5, baseWidth * 0.5, targetAlpha * 0.3, profile);
        break;
    }

    if (this.debugMode) {
      this.drawDebugFrame(ctx, baseLength, baseWidth, input.classKey);
    }

    ctx.restore();
  }

  private drawLinearWake(
    ctx: CanvasRenderingContext2D, 
    len: number, 
    width: number, 
    alpha: number, 
    prof: ActiveWakePolishProfile
  ): void {
    // Generate soft linear pathing vectors stretching downstream from stern
    const grad = ctx.createLinearGradient(0, 0, -len, 0);
    grad.addColorStop(0.0, "rgba(240, 248, 255, 0.0)");
    grad.addColorStop(0.15, `rgba(240, 248, 255, ${alpha * 0.8})`);
    grad.addColorStop(0.35, `rgba(240, 248, 255, ${alpha})`);
    grad.addColorStop(0.70, `rgba(240, 248, 255, ${alpha * 0.3})`);
    grad.addColorStop(1.0, "rgba(240, 248, 255, 0.0)");

    ctx.beginPath();
    ctx.moveTo(0, 0);
    ctx.lineTo(-len, 0);
    ctx.strokeStyle = grad;
    ctx.lineWidth = width;
    ctx.shadowBlur = prof.lineSoftness;
    ctx.shadowColor = "rgba(240, 248, 255, 0.2)";
    ctx.stroke();
  }

  private drawSplitVWake(
    ctx: CanvasRenderingContext2D, 
    len: number, 
    width: number, 
    alpha: number, 
    prof: ActiveWakePolishProfile
  ): void {
    const angleRad = (14 * Math.PI) / 180; // Standardized V-parting angle bounds
    
    ctx.save();
    ctx.shadowBlur = prof.lineSoftness;
    ctx.shadowColor = "rgba(240, 248, 255, 0.15)";

    for (const side of [1, -1]) {
      const grad = ctx.createLinearGradient(0, 0, -len, side * len * Math.sin(angleRad));
      grad.addColorStop(0.0, "rgba(240, 248, 255, 0.0)");
      grad.addColorStop(0.2, `rgba(240, 248, 255, ${alpha})`);
      grad.addColorStop(0.6, `rgba(240, 248, 255, ${alpha * 0.4})`);
      grad.addColorStop(1.0, "rgba(240, 248, 255, 0.0)");

      ctx.beginPath();
      ctx.moveTo(0, 0);
      ctx.lineTo(-len, side * len * Math.sin(angleRad));
      ctx.strokeStyle = grad;
      ctx.lineWidth = width * 0.8;
      ctx.stroke();
    }
    ctx.restore();
  }

  private drawTurbulentWake(
    ctx: CanvasRenderingContext2D, 
    len: number, 
    width: number, 
    alpha: number, 
    vesselId: string,
    prof: ActiveWakePolishProfile
  ): void {
    // Generate deterministic variations using a basic non-state hashing method
    const seed = this.hashString(vesselId);
    const count = Math.min(prof.turbulenceCount, MAX_TURBULENCE_FILAMENTS);

    ctx.save();
    for (let i = 0; i < count; i++) {
      const offsetPhase = (seed + i * 31) % 100 / 100;
      const jitterY = (Math.sin(offsetPhase * Math.PI * 2) * width * 0.4 * prof.turbulenceSpread);
      const filamentLen = len * (0.4 + ((seed + i * 7) % 50 / 100) * prof.turbulenceLengthScale);

      const grad = ctx.createLinearGradient(0, 0, -filamentLen, jitterY);
      grad.addColorStop(0.0, `rgba(230, 245, 255, ${alpha * prof.nearSternAlpha})`);
      grad.addColorStop(0.4, `rgba(230, 245, 255, ${alpha * 0.7})`);
      grad.addColorStop(1.0, "rgba(230, 245, 255, 0.0)");

      ctx.beginPath();
      ctx.moveTo(0, 0);
      ctx.quadraticCurveTo(-filamentLen * 0.4, jitterY * 1.5, -filamentLen, jitterY);
      ctx.strokeStyle = grad;
      ctx.lineWidth = width * 0.6;
      ctx.stroke();
    }
    ctx.restore();
  }

  private drawDriftWake(
    ctx: CanvasRenderingContext2D, 
    len: number, 
    width: number, 
    alpha: number, 
    prof: ActiveWakePolishProfile
  ): void {
    // Structural asymmetry simulation rule: bias drawing metrics along the left port axis
    const grad = ctx.createLinearGradient(0, 0, -len, -width * 0.5);
    grad.addColorStop(0.0, "rgba(240, 248, 255, 0.0)");
    grad.addColorStop(0.25, `rgba(240, 248, 255, ${alpha})`);
    grad.addColorStop(0.75, `rgba(240, 248, 255, ${alpha * 0.2})`);
    grad.addColorStop(1.0, "rgba(240, 248, 255, 0.0)");

    ctx.beginPath();
    ctx.moveTo(0, 0);
    ctx.quadraticCurveTo(-len * 0.5, -width * 0.8, -len, -width * 0.5);
    ctx.strokeStyle = grad;
    ctx.lineWidth = width * 1.1;
    ctx.stroke();
  }

  private drawDebugFrame(ctx: CanvasRenderingContext2D, len: number, width: number, label: string): void {
    ctx.strokeStyle = "rgba(255, 0, 0, 0.65)";
    ctx.lineWidth = 1.0;
    ctx.strokeRect(0, -width/2, -len, width);
    ctx.fillStyle = "rgba(255, 0, 0, 0.85)";
    ctx.font = "9px monospace";
    ctx.fillText(`${label.toUpperCase()} WAKE BOUNDS`, -len, -width);
  }

  private hashString(str: string): number {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      hash = str.charCodeAt(i) + ((hash << 5) - hash);
    }
    return Math.abs(hash);
  }
}
