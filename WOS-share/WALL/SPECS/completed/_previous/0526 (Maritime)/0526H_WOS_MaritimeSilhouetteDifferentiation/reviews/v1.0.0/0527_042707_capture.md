# 1. CORE ARCHITECTURAL DOCTRINE

The `MaritimeSilhouetteDifferentiation` module translates tabular tracking metrics into a perceptual hierarchy of shape, weight, and motion inertia across the harbor canvas.

- **Presentation Interprets Reality:** Silhouette transformation exists exclusively within the 2.5D presentation space. It modifies visual presentation fields but is strictly forbidden from editing underlying simulation parameters.
- **Identity Through Shape Language:** WOS completely rejects literal vessel renderings. Maritime presence emerges through structural implication: aspect ratios, light spacing, mass distributions, and wake dynamics.
- **The Zero-Texture Directive:** The engine generates all structural envelopes on-the-fly via mathematical drawing routines. Photo textures, sprite atlases, and pre-baked geometry are permanently banned from the runtime landscape.
- **Diagnostic Immunity:** Validation and system-level debug vessels bypass all atmospheric degradation, structural warping, and drift filtering to ensure absolute measurement accuracy during internal analysis.

export class MaritimeSilhouetteDifferentiationModule {
  constructor() {}

  public mapVesselToSilhouetteClass(vesselClass: string, speedKts: number): MaritimeSilhouetteClass {
    const cls = vesselClass?.toUpperCase() || "UNKNOWN";

    if (cls === "ANCHORED" || speedKts < 0.1) return "ANCHORED";
    if (cls === "CARGO" || cls === "TANKER") return "CARGO";
    if (cls === "FERRY" || cls === "PASSENGER") return "FERRY";
    if (cls === "TUG" || cls === "INDUSTRIAL" || cls === "FISHING") return "TUG";
    if (cls === "FAST_CRAFT" || cls === "RECREATIONAL" || cls === "SERVICE") return "FAST_CRAFT";
    
    return "UNKNOWN";
  }

  public resolveSilhouetteProfile(input: MaritimeSilhouetteInput): MaritimeSilhouetteProfile {
    // 1. Diagnostic Verification Path Bypass
    if (input.isValidationEntity) {
      return {
        version: "1.0.0", silhouetteClass: "UNKNOWN",
        hullAspectBias: 1.5, atmosphericMassBias: 0.0,
        wakePersistence: 1.0, directionalConfidence: 1.0, turnSoftnessDeg: 0.0,
        farLightSpacing: 1.0, lightClusterVariance: 0.0, bloomSoftness: 0.0
      };
    }

    // 2. Class Resolution Mapping
    let targetClass = this.mapVesselToSilhouetteClass(input.vesselClass, input.speedKts);
    
    // 3. Process Distance Band Degradation Rules
    const band = input.distanceBand || "NEAR";
    
    if (band === "FAR") {
      targetClass = "DISTANT_TRAFFIC";
    } else if (band === "ATMOSPHERIC" || input.visibilityClass === "LIGHT_ONLY") {
      // Hull dimensions scale completely out; handing structural presence over to point signals
      return {
        version: "1.0.0", silhouetteClass: "DISTANT_TRAFFIC",
        hullAspectBias: 0.0, atmosphericMassBias: 0.0,
        wakePersistence: 0.0, directionalConfidence: 0.90, turnSoftnessDeg: 10.0,
        farLightSpacing: 1.0, lightClusterVariance: 0.10, bloomSoftness: 0.50
      };
    }

    const baseProfile = SILHOUETTE_PROFILES[targetClass] || SILHOUETTE_PROFILES["UNKNOWN"];

    // 4. Evaluate Shape Modifications Based on Parental Visibility Class Constraints
    let finalAspect = baseProfile.hullAspectBias;
    let finalMass = baseProfile.atmosphericMassBias;
    let finalWake = baseProfile.wakePersistence;

    const vClass = input.visibilityClass?.toUpperCase() || "FULL";
    if (vClass === "REDUCED") {
      finalAspect *= 0.85;
      finalWake *= 0.70;
    } else if (vClass === "SILHOUETTE") {
      finalMass *= 1.25; // Boost geometric emphasis for high-contrast presentation profiles
      finalWake *= 0.40;
    } else if (vClass === "MARKER_ONLY") {
      finalAspect = 0.1;
      finalWake = 0.0;
    } else if (vClass === "ATMOSPHERIC_HIDDEN") {
      finalAspect = 0.0;
      finalMass = 0.0;
      finalWake = 0.0;
    }

    return {
      version: "1.0.0",
      silhouetteClass: targetClass,
      hullAspectBias: finalAspect,
      atmosphericMassBias: finalMass,
      wakePersistence: finalWake,
      directionalConfidence: baseProfile.directionalConfidence,
      turnSoftnessDeg: baseProfile.turnSoftnessDeg,
      farLightSpacing: baseProfile.farLightSpacing,
      lightClusterVariance: baseProfile.lightClusterVariance,
      bloomSoftness: baseProfile.bloomSoftness
    };
  }
}
