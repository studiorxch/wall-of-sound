export class MaritimeSilhouetteDifferentiationModule {
  constructor() {}

  public resolveSilhouetteClass(vesselClass: string, vesselState?: string | null): MaritimeSilhouetteClass {
    const cls = (vesselClass?.toLowerCase() || "unknown").trim();
    
    // Exact structural map check
    if (cls === "cargo") return "cargo";
    if (cls === "tanker") return "tanker";
    if (cls === "ferry") return "ferry";
    if (cls === "service") return "service";
    if (cls === "recreational") return "recreational";
    if (cls === "fishing") return "fishing";
    if (cls === "passenger") return "passenger";
    if (cls === "tug") return "tug";
    if (cls === "military") return "military";
    if (cls === "industrial") return "industrial";

    // Structural Redirection Rules
    if (cls === "cruise") return "passenger";
    if (cls === "container" || cls === "freighter") return "cargo";
    if (cls === "barge") return "industrial";
    if (cls === "yacht" || cls === "speedboat") return "recreational";
    if (cls === "trawler") return "fishing";
    if (cls === "pilot" || cls === "patrol") return "service";

    return "unknown";
  }

  public resolveSilhouetteProfile(input: MaritimeSilhouetteInput): MaritimeSilhouetteProfile {
    // 1. Diagnostic Verification Path Bypass (Validation Immunity)
    if (input.isValidationEntity) {
      return {
        version: "1.0.1",
        silhouetteClass: "unknown",
        hullAspectBias: 1.50,
        atmosphericMassBias: 1.00,
        wakeReadabilityScale: 1.00,
        headingStabilityBias: 1.00,
        turnSoftnessDeg: 0.0,
        farLightSpacing: 1.00,
        lightClusterVariance: 0.00,
        bloomSoftness: 0.00
      };
    }

    // 2. Class Resolution Mapping
    const resolvedClass = this.resolveSilhouetteClass(input.vesselClass, input.vesselState);
    const baseProfile = SILHOUETTE_PROFILES[resolvedClass] || SILHOUETTE_PROFILES["unknown"];

    // 3. Process Dynamic State Modification
    let aspectMod = 1.0;
    let wakeMod = 1.0;
    
    const state = (input.vesselState?.toLowerCase() || "underway").trim();
    if (state === "anchored" || state === "moored" || input.speedKts < 0.15) {
      aspectMod = 1.05; // Ground hull shape presentation when locked
      wakeMod = 0.00;   // Wake generation trails scale out instantly
    }

    // 4. Implement Distance Collapse and Visibility Constraints
    let finalAspect = baseProfile.hullAspectBias * aspectMod;
    let finalMass = baseProfile.atmosphericMassBias;
    let finalWake = baseProfile.wakeReadabilityScale * wakeMod;
    let finalLightSpacing = baseProfile.farLightSpacing;

    const band = input.distanceBand || "NEAR";

    if (band === "MID") {
      finalAspect *= 0.80;
      finalWake *= 0.65;
    } else if (band === "FAR") {
      finalAspect *= 0.25; // Hull presentation scales down toward zero bounds
      finalWake *= 0.15;
      finalLightSpacing *= 1.20; // Light clusters spread out horizontally to preserve recognition profile
    } else if (band === "ATMOSPHERIC") {
      // Light spacing takes over entirely as geometric hull features zero out
      return {
        version: "1.0.1", silhouetteClass: resolvedClass,
        hullAspectBias: 0.00, atmosphericMassBias: 0.00,
        wakeReadabilityScale: 0.00, headingStabilityBias: 0.95, turnSoftnessDeg: 5.0,
        farLightSpacing: baseProfile.farLightSpacing * 1.50,
        lightClusterVariance: baseProfile.lightClusterVariance * 1.40,
        bloomSoftness: baseProfile.bloomSoftness * 1.30
      };
    }

    // Apply upstream parental visibility classes
    const vClass = input.visibilityClass || "FULL";
    if (vClass === "REDUCED") {
      finalAspect *= 0.85;
      finalWake *= 0.50;
    } else if (vClass === "SILHOUETTE") {
      finalMass = Math.min(finalMass * 1.30, 1.00); // Increase edge contrast to cut through haze
      finalWake *= 0.25;
    } else if (vClass === "MARKER_ONLY") {
      finalAspect = 0.05;
      finalWake = 0.00;
    } else if (vClass === "LIGHT_ONLY" || vClass === "ATMOSPHERIC_HIDDEN") {
      finalAspect = 0.00;
      finalMass = 0.00;
      finalWake = 0.00;
    }

    // 5. Final Safety Guard Checks
    finalAspect = Math.min(Math.max(finalAspect, 0.00), MAX_ASPECT_BIAS);

    return {
      version: "1.0.1",
      silhouetteClass: resolvedClass,
      hullAspectBias: finalAspect,
      atmosphericMassBias: finalMass,
      wakeReadabilityScale: finalWake,
      headingStabilityBias: baseProfile.headingStabilityBias,
      turnSoftnessDeg: baseProfile.turnSoftnessDeg,
      farLightSpacing: finalLightSpacing,
      lightClusterVariance: baseProfile.lightClusterVariance,
      bloomSoftness: baseProfile.bloomSoftness
    };
  }

  public getSilhouetteConstants(): object {
    return { MARITIME_SILHOUETTE_VERSION, MIN_ASPECT_BIAS, MAX_ASPECT_BIAS, DEFAULT_STABILITY_BIAS, MAX_TURN_SOFTNESS_DEG };
  }
}
