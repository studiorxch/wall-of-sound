# 1. CORE ARCHITECTURAL DOCTRINE

The `MaritimeWaterMemory` system tracks and renders historical vessel motion fields by combining past wake footprints into smooth visual paths on the water's surface.

- **Water remembers motion briefly. It does NOT simulate water physics.**
- **Presentation-Layer Only:** Memory patterns display where activity has recently occurred, but they do not reflect true absolute fluid distributions.
- **2D owns truth. 2.5D owns presentation.**
- **No Entity Fabrication:** Historic data fields cannot generate active tracker records or imply that a hidden vessel exists.
- **Strict Degradation:** All motion artifacts decay predictably toward zero intensity according to exponential runtime constants.

---

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Accumulating historical wake footprint data using discrete screen-space quantization grids.
- Computing localized cell degradation using standard exponential half-life formulas.
- Storing layout parameters for visual archetypes like `WAKE_SPINE`, `WAKE_ARM`, and `CHURN`.
- Applying clutter-based blending overrides to smooth out visualization layouts in dense traffic zones.
- Exporting real-time metric snapshots and diagnostic tracking snapshots.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Active vessel data records, positional indices, or update intervals inside `AISRuntime`.
- Active wake lines or live propagation vectors governed by `MaritimeWakeAuthority`.
- Target priorities, visibility modifiers, or tier categories within `MaritimePopulationHierarchy`.
- Base atmospheric tracking fields defined inside `VisibilityClassRuntime`.
- Global color themes, canvas dimensions, or layout transforms managed by `MarineRenderer`.
- View parameters, viewport orientations, or focus paths within camera modules.

  private stamps: WaterMemoryStamp[] = [];
  private cells: Map<string, WaterMemoryCell> = new Map();
  private globalActiveFlag: boolean = true;

  constructor() {}

  private generateCellKey(gridX: number, gridY: number): string {
    return `${gridX}_${gridY}`;
  }

  public normalizeClass(rawClass: string | null | undefined): WaterMemoryClass {
    if (!rawClass) return "UNKNOWN";
    const clean = rawClass.toUpperCase().trim();
    const list: WaterMemoryClass[] = [
      "CARGO", "TANKER", "FERRY", "TUG", "RECREATIONAL", 
      "FISHING", "PASSENGER", "MILITARY", "INDUSTRIAL", "SERVICE"
    ];
    return list.includes(clean as WaterMemoryClass) ? (clean as WaterMemoryClass) : "UNKNOWN";
  }

  private wakeModeToStampKind(mode: string): WaterMemoryStampKind {
    switch (mode?.toUpperCase()) {
      case "SPINE": return "WAKE_SPINE";
      case "ARM": return "WAKE_ARM";
      case "TURBULENCE": return "TURBULENCE";
      case "DRIFT": return "DRIFT";
      case "CHURN": return "CHURN";
      default: return "WAKE_SPINE";
    }
  }

  public stampWakeMemory(input: WaterMemoryStampInput): void {
    if (!this.globalActiveFlag) return;
    if (input.speedKts < MIN_STAMP_SPEED_KTS) return;

    const now = input.nowMs ?? performance.now();
    const vClass = this.normalizeClass(input.vesselClass);
    const kind = this.wakeModeToStampKind(input.wakeMode);

    const baseIntensity = input.intensity ?? 0.6;
    
    const newStamp: WaterMemoryStamp = {
      stampId: `stmp_${now}_${Math.random().toString(36).substr(2, 4)}`,
      vesselId: input.vesselId || null,
      vesselClass: vClass,
      kind,
      x: input.x,
      y: input.y,
      headingDeg: input.headingDeg,
      lengthPx: input.lengthPx,
      widthPx: input.widthPx,
      intensity: Math.min(Math.max(baseIntensity, 0), 1),
      persistenceMs: DEFAULT_DECAY_HALF_LIFE_MS,
      createdAtMs: now,
      seed: input.seed || Math.floor(Math.random() * 1000)
    };

    this.stamps.push(newStamp);
    if (this.stamps.length > MAX_ACTIVE_STAMPS) {
      this.stamps.shift();
    }

    // Direct quantization into screen-space cells
    const gridX = Math.floor(input.x / DEFAULT_CELL_SIZE_PX);
    const gridY = Math.floor(input.y / DEFAULT_CELL_SIZE_PX);
    const cellKey = this.generateCellKey(gridX, gridY);

    const rad = (input.headingDeg * Math.PI) / 180;
    const hX = Math.cos(rad);
    const hY = Math.sin(rad);

    const existingCell = this.cells.get(cellKey);
    if (existingCell) {
      const combinedIntensity = Math.min(existingCell.intensity + baseIntensity * 0.4, 1.0);
      const isNewDominant = baseIntensity > existingCell.intensity;

      this.cells.set(cellKey, {
        cellId: existingCell.cellId,
        x: gridX * DEFAULT_CELL_SIZE_PX + DEFAULT_CELL_SIZE_PX / 2,
        y: gridY * DEFAULT_CELL_SIZE_PX + DEFAULT_CELL_SIZE_PX / 2,
        intensity: combinedIntensity,
        ageMs: 0,
        dominantClass: isNewDominant ? vClass : existingCell.dominantClass,
        dominantKind: isNewDominant ? kind : existingCell.dominantKind,
        headingVectorX: isNewDominant ? hX : existingCell.headingVectorX,
        headingVectorY: isNewDominant ? hY : existingCell.headingVectorY,
        churn: kind === "TURBULENCE" || kind === "CHURN" ? Math.min(existingCell.churn + 0.3, 1.0) : existingCell.churn,
        lastUpdatedMs: now
      });
    } else {
      if (this.cells.size >= MAX_ACTIVE_CELLS) {
        const oldestKey = this.cells.keys().next().value;
        if (oldestKey) this.cells.delete(oldestKey);
      }

      this.cells.set(cellKey, {
        cellId: `cell_${cellKey}`,
        x: gridX * DEFAULT_CELL_SIZE_PX + DEFAULT_CELL_SIZE_PX / 2,
        y: gridY * DEFAULT_CELL_SIZE_PX + DEFAULT_CELL_SIZE_PX / 2,
        intensity: baseIntensity,
        ageMs: 0,
        dominantClass: vClass,
        dominantKind: kind,
        headingVectorX: hX,
        headingVectorY: hY,
        churn: kind === "TURBULENCE" || kind === "CHURN" ? 0.5 : 0.0,
        lastUpdatedMs: now
      });
    }
  }

  public updateWaterMemory(deltaMs: number, nowMs: number): void {
    // Purge expired history stamps
    this.stamps = this.stamps.filter(s => nowMs - s.createdAtMs < MAX_PERSISTENCE_MS);

    // Exponentially decay localized cell values
    for (const [key, cell] of this.cells.entries()) {
      const currentAge = nowMs - cell.lastUpdatedMs;
      const decayFactor = Math.pow(0.5, deltaMs / DEFAULT_DECAY_HALF_LIFE_MS);
      const updatedIntensity = cell.intensity * decayFactor;

      if (updatedIntensity < DISCARD_THRESHOLD) {
        this.cells.delete(key);
      } else {
        this.cells.set(key, {
          ...cell,
          intensity: updatedIntensity,
          ageMs: currentAge
        });
      }
    }
  }

  public renderWaterMemory(ctx: CanvasRenderingContext2D, options: WaterMemoryRenderOptions): void {
    if (options.isAtmosphericSuppressed || this.cells.size === 0) return;

    ctx.save();
    for (const cell of this.cells.values()) {
      if (cell.dominantKind === "LANE" && !options.showLanes) continue;
      if ((cell.dominantKind === "TURBULENCE" || cell.dominantKind === "CHURN") && !options.showChurn) continue;

      const alpha = cell.intensity * options.globalAlphaModifier * 0.25;
      ctx.beginPath();
      
      // Compute subtle historical direction indicators using vector fields
      const strokeLength = DEFAULT_CELL_SIZE_PX * (0.5 + cell.intensity * 0.5);
      const startX = cell.x - (cell.headingVectorX * strokeLength) / 2;
      const startY = cell.y - (cell.headingVectorY * strokeLength) / 2;
      const endX = cell.x + (cell.headingVectorX * strokeLength) / 2;
      const endY = cell.y + (cell.headingVectorY * strokeLength) / 2;

      ctx.moveTo(startX, startY);
      ctx.lineTo(endX, endY);

      // Map class identifiers to structural color states
      ctx.strokeStyle = this.getClassColorToken(cell.dominantClass);
      ctx.lineWidth = 4.0 + cell.churn * 6.0;
      ctx.globalAlpha = alpha;
      ctx.stroke();
    }
    ctx.restore();
  }

  private getClassColorToken(vClass: WaterMemoryClass): string {
    switch (vClass) {
      case "CARGO": return "rgba(70, 100, 120, 0.6)";
      case "FERRY": return "rgba(0, 200, 255, 0.7)";
      case "TUG": return "rgba(255, 130, 0, 0.6)";
      default: return "rgba(255, 255, 255, 0.4)";
    }
  }

  public clearWaterMemory(): void {
    this.stamps = [];
    this.cells.clear();
  }

  public getCells(): readonly WaterMemoryCell[] {
    return Array.from(this.cells.values());
  }

  public getWaterMemorySnapshot(): WaterMemorySnapshot {
    let sum = 0;
    let max = 0;
    for (const cell of this.cells.values()) {
      sum += cell.intensity;
      if (cell.intensity > max) max = cell.intensity;
    }

    return {
      version: "1.0.0",
      active: this.globalActiveFlag,
      stampCount: this.stamps.length,
      cellCount: this.cells.size,
      totalIntensity: sum,
      maxIntensity: max,
      dominantClass: this.stamps[this.stamps.length - 1]?.vesselClass || null,
      lastUpdateMs: Date.now()
    };
  }
}
