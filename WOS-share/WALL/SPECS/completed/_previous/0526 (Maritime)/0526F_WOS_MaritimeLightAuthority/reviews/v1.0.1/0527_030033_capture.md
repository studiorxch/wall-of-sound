# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Generating the master, immutable `MaritimeLightEnvelope` structures.
- Regulating bloom radius boundaries, transparency properties, and light dissipation rules.
- Defining color profiles and cluster separation spreads ($\text{clusterSpreadPx}$) across all vessel types.
- Calculating deterministic, phase-aligned light pulsing behaviors based on entity attributes.
- Evaluating the atmospheric reduction formula ($\text{suppressUnderAtmosphere}$) against incoming conditions.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Underlying telemetry data, speed variables, true headings, or identification indices in `AISRuntime`.
- Active display configurations or global viewport layouts managed by the engine camera.
- Positional matrices, level-of-detail triggers, or trailing paths owned by `ActiveWakePolish`.
- Global color registries or master palette lists inside `MaritimeStyleRegistry`.

6. EXECUTABLE LIGHT RESOLUTION ENGINETypeScriptexport class MaritimeLightAuthorityModule {
  constructor() {}

  public resolveClassLightSignature(vesselClass: string): MaritimeClassLightSignature {
    const key = (vesselClass?.toLowerCase() || "unknown") as MaritimeClassKey;
    // v1.0.x Cruise Redirection Boundary Fallback Rules
    if (key === ("cruise" as any)) return CLASS_LIGHT_SIGNATURES["passenger"];
    return CLASS_LIGHT_SIGNATURES[key] || CLASS_LIGHT_SIGNATURES["unknown"];
  }

  private generateStableSeed(input: MaritimeLightInput): number {
    const primaryStr = String(input.mmsi || input.vesselId || input.vesselClass || "0");
    let hash = 0;
    for (let i = 0; i < primaryStr.length; i++) {
      hash = primaryStr.charCodeAt(i) + ((hash << 5) - hash);
    }
    return Math.abs(hash);
  }

  public resolveLightEnvelope(input: MaritimeLightInput): MaritimeLightEnvelope {
    // 1. Input Verification Check
    if (!input || !input.vesselClass) {
      return this.getFallbackLightEnvelope("FALLBACK_INVALID_INPUT");
    }

    // 2. Parental Overrides: Structural Visibility Suppressions
    if (input.visibilityClass === "ATMOSPHERIC_HIDDEN") {
      return this.getFallbackLightEnvelope("VISIBILITY_SUPPRESSED");
    }

    const signature = this.resolveClassLightSignature(input.vesselClass);
    const seed = this.generateStableSeed(input);

    // 3. Evaluate Deterministic Waveforms (Ambient Pulses & High-Frequency Shimmer)
    const pulsePhase = (seed % 360) * (Math.PI / 180);
    const timeSec = input.nowMs / 1000;
    
    // Canonical Pulse Rule Formulation
    const pulseVal = 1.0 - signature.pulseDepth + 
      signature.pulseDepth * (0.5 + 0.5 * Math.sin(timeSec * signature.pulseHz * Math.PI * 2 + pulsePhase));

    const shimmerNoise = (Math.sin(timeSec * 31.4 + (seed % 19)) * 0.5 + 0.5 * Math.cos(timeSec * 51.2 + (seed % 29)));
    const activeShimmer = DEFAULT_SHIMMER_AMOUNT * signature.shimmerScale * shimmerNoise;

    // 4. Evaluate Step-Down Mode Collapse Boundaries Based on Distance Bands
    let targetMode = signature.baseRenderMode;
    let reason: MaritimeLightReasonCode = "LIGHT_POINT";
    let distanceBand = "NEAR";

    if (input.distanceEnvelope) {
      distanceBand = input.distanceEnvelope.band;
      
      if (distanceBand === "MID") {
        if (targetMode === "DUAL_NAV") targetMode = "CLUSTER";
      } else if (distanceBand === "FAR") {
        targetMode = (targetMode === "GHOST") ? "GHOST" : "POINT";
      } else if (distanceBand === "ATMOSPHERIC") {
        targetMode = "GHOST";
      }
    }

    // 5. Connect and Process Visibility Constraints
    if (input.visibilityClass) {
      const vClass = input.visibilityClass.toUpperCase().trim();
      if (vClass === "LIGHT_ONLY" || vClass === "MARKER_ONLY" || vClass === "SILHOUETTE") {
        if (targetMode !== "GHOST") targetMode = "POINT";
      } else if (vClass === "REDUCED") {
        if (targetMode === "DUAL_NAV") targetMode = "CLUSTER";
      }
    }

    // Assign explicit tracking codes based on finalized rendering styles
    if (targetMode === "DUAL_NAV") reason = "LIGHT_DUAL_NAV";
    else if (targetMode === "CLUSTER") reason = "LIGHT_CLUSTER";
    else if (targetMode === "POINT") reason = "LIGHT_POINT";
    else if (targetMode === "GHOST") reason = "LIGHT_GHOST";
    else {
      return this.getFallbackLightEnvelope("LIGHT_SUPPRESSED");
    }

    // 6. Execute Mathematical Atmosphere Suppression Reductions
    const fog = input.fogAlpha ?? 0.0;
    const haze = input.hazeAlpha ?? 0.0;
    const density = input.densityPressure ?? 0.0;
    const atmospherePressure = Math.min(Math.max((fog * 0.45) + (haze * 0.35) + (density * 0.25), 0), 1);
    
    let finalAlpha = signature.baseAlpha * pulseVal;
    finalAlpha *= (1.0 - atmospherePressure * signature.suppressUnderAtmosphere);

    // Apply distance modifications derived directly from the active envelope
    if (input.distanceEnvelope) {
      finalAlpha *= input.distanceEnvelope.lightAlpha;
    }
    
    finalAlpha = Math.min(Math.max(finalAlpha + activeShimmer, MIN_LIGHT_ALPHA), MAX_NAV_ALPHA);

    if (finalAlpha <= MIN_LIGHT_ALPHA + 0.001) {
      return this.getFallbackLightEnvelope("DISTANCE_SUPPRESSED");
    }

    // 7. Calculate Color Profiles and Warmth Distributions
    const navPortColor = `rgba(235, 64, 52, ${finalAlpha})`;
    const navStarboardColor = `rgba(52, 235, 110, ${finalAlpha})`;
    const navSternColor = `rgba(245, 245, 240, ${finalAlpha * 0.95})`;
    
    const warmth = signature.farWarmth;
    const rGlow = Math.floor(240 * (1.0 - warmth) + 255 * warmth);
    const gGlow = Math.floor(242 * (1.0 - warmth) + 175 * warmth);
    const bGlow = Math.floor(255 * (1.0 - warmth) + 45 * warmth);
    const glowColor = `rgba(${rGlow}, ${gGlow}, ${bGlow}, ${finalAlpha * 0.35})`;

    // 8. Compute Structural Edge Bloom Scaling
    const zoomScale = input.zoom / 13.0;
    let baseBloomRadius = 4.0 * signature.bloomScale * zoomScale;
    if (distanceBand === "FAR" || distanceBand === "ATMOSPHERIC") {
      baseBloomRadius *= 1.35; // Far-range lights expand slightly but dim out softly
    }
    const finalBloomRadius = Math.min(Math.max(baseBloomRadius, MIN_BLOOM_RADIUS_PX), MAX_BLOOM_RADIUS_PX);
    const finalBloomAlpha = Math.min(MAX_BLOOM_ALPHA * finalAlpha * (1.0 - fog), MAX_BLOOM_ALPHA);

    // 9. Build Immutable Presentation Struct Output
    return {
      version: "1.0.1",
      visible: true,
      renderMode: targetMode,
      reasonCode: reason,
      alpha: finalAlpha,
      bloomAlpha: finalBloomAlpha,
      bloomRadiusPx: finalBloomRadius,
      navAlpha: (targetMode === "DUAL_NAV") ? finalAlpha : (targetMode === "CLUSTER" ? finalAlpha * 0.7 : 0),
      farAlpha: (targetMode === "POINT" || targetMode === "GHOST") ? finalAlpha * 0.85 : 0,
      pulsePhase,
      pulseValue: pulseVal,
      shimmerAmount: activeShimmer,
      navPortColor,
      navStarboardColor,
      navSternColor,
      glowColor,
      allowNavPair: targetMode === "DUAL_NAV",
      allowMastLight: targetMode === "DUAL_NAV" || targetMode === "CLUSTER",
      allowFarGlint: targetMode === "POINT" || targetMode === "GHOST",
      allowBloom: finalBloomAlpha > 0.01 && targetMode !== "GHOST",
      allowWakeGlow: targetMode === "DUAL_NAV" && distanceBand === "HERO",
      allowReflectionHint: distanceBand === "HERO" || distanceBand === "NEAR"
    };
  }

  public getFallbackLightEnvelope(reasonCode: MaritimeLightReasonCode): MaritimeLightEnvelope {
    const isDist = reasonCode === "DISTANCE_SUPPRESSED";
    return {
      version: "1.0.1",
      visible: isDist,
      renderMode: isDist ? "GHOST" : "NONE",
      reasonCode,
      alpha: isDist ? 0.03 : 0,
      bloomAlpha: 0,
      bloomRadiusPx: 0,
      navAlpha: 0,
      farAlpha: isDist ? 0.03 : 0,
      pulsePhase: 0,
      pulseValue: 0,
      shimmerAmount: 0,
      navPortColor: "rgba(0,0,0,0)",
      navStarboardColor: "rgba(0,0,0,0)",
      navSternColor: "rgba(0,0,0,0)",
      glowColor: "rgba(0,0,0,0)",
      allowNavPair: false,
      allowMastLight: false,
      allowFarGlint: isDist,
      allowBloom: false,
      allowWakeGlow: false,
      allowReflectionHint: false
    };
  }

  public getConstants(): object {
    return { MARITIME_LIGHT_AUTHORITY_VERSION, MIN_LIGHT_ALPHA, MAX_NAV_ALPHA, MAX_FAR_LIGHT_ALPHA };
  }
}
