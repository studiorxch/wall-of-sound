# 1. CORE ARCHITECTURAL DOCTRINE

The `MaritimeLightAuthority` layer governs the rendering, bloom adjustments, and distinct color signatures of vessel lights across the harbor canvas.

- **At distance, lights become the vessel.**
- **Purely Interpretive Behavior:** The module does not manage electrical systems or simulation logic. It is a rendering filter that translates tracking metrics into visual effects.
- **2D owns truth. 2.5D owns presentation.**
- **Zero-Randomness Rule:** All visual pulsing, shimmer, and layout shifts must be computed using deterministic formulas tied to stable entity tracking numbers. The use of runtime random functions is completely prohibited.
- **Strict Visual Restraint:** Lights indicate presence; they are completely barred from using flashing frequencies or warning indicators that mimic emergency vehicles.

---

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Defining the 5 primary display styles (`DUAL_NAV`, `CLUSTER`, `POINT`, `GHOST`, `NONE`) across all 11 core ship categories.
- Managing localized spatial coordinates for navigation lights and cluster configurations.
- Controlling mathematical pulse rates, depth values, and asynchronous phase adjustments.
- Calculating edge bloom boundaries and light dissipation values across changing distances.
- Restricting light indicators based on input from upstream visibility layers.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Absolute location vectors, speed metrics, tracking headings, or status tags within `AISRuntime`.
- Base distance calculations or visibility-band assignments handled by `MaritimeDistanceAtmosphere`.
- Regional weather patterns, fog tracking, or visibility flags owned by `VisibilityClassRuntime`.
- Global look-up tables or color indices managed within the `MaritimeStyleRegistry`.

6. EXECUTABLE PURE ILLUMINATION ENGINETypeScriptexport class MaritimeLightAuthorityModule {
  constructor() {}

  public resolveClassLightSignature(vesselClass: string): MaritimeClassLightSignature {
    const key = vesselClass?.toLowerCase() || "unknown";
    return CLASS_LIGHT_SIGNATURES[key] || CLASS_LIGHT_SIGNATURES["unknown"];
  }

  private generateStableSeed(input: MaritimeLightInput): number {
    const primaryStr = String(input.mmsi || input.vesselId || input.vesselClass || "0");
    let hash = 0;
    for (let i = 0; i < primaryStr.length; i++) {
      hash = primaryStr.charCodeAt(i) + ((hash << 5) - hash);
    }
    return Math.abs(hash);
  }

  public resolveLightEnvelope(input: MaritimeLightInput): MaritimeLightEnvelope {
    // 1. Boundary Verification Fallback Check
    if (!input || !input.vesselClass) {
      return this.getFallbackLightEnvelope("FALLBACK_INVALID_INPUT");
    }

    // 2. Upstream Parental Visibility Hierarchy Suppression Check
    if (input.visibilityClass === "ATMOSPHERIC_HIDDEN") {
      return this.getFallbackLightEnvelope("VISIBILITY_SUPPRESSED");
    }

    const signature = this.resolveClassLightSignature(input.vesselClass);
    const seed = this.generateStableSeed(input);

    // 3. Evaluate Deterministic Sinusoidal Temporal Waveforms
    const pulsePhase = (seed % 360) * (Math.PI / 180);
    const timeSec = input.nowMs / 1000;
    const pulseVal = 1.0 - signature.pulseDepth + 
      signature.pulseDepth * (0.5 + 0.5 * Math.sin(timeSec * signature.pulseHz * Math.PI * 2 + pulsePhase));

    // Evaluate high-frequency deterministic shimmer variance
    const shimmerNoise = (Math.sin(timeSec * 25.0 + (seed % 17)) * 0.5 + 0.5 * Math.cos(timeSec * 42.0 + (seed % 23)));
    const activeShimmer = DEFAULT_SHIMMER_AMOUNT * signature.shimmerScale * shimmerNoise;

    // 4. Connect and Reconcile with the Distance Envelope Bounds
    let targetMode = signature.baseRenderMode;
    let reason: MaritimeLightReasonCode = "LIGHT_POINT";
    let distMultiplier = 1.0;
    let distanceBand = "NEAR";

    if (input.distanceEnvelope) {
      distanceBand = input.distanceEnvelope.band;
      distMultiplier = input.distanceEnvelope.lightAlpha;

      // Force render mode simplification based on clear distance boundaries
      if (distanceBand === "FAR") {
        targetMode = (signature.baseRenderMode === "GHOST") ? "GHOST" : "POINT";
      } else if (distanceBand === "ATMOSPHERIC") {
        targetMode = "GHOST";
      }
    }

    if (distMultiplier <= 0.01 || targetMode === "NONE") {
      return this.getFallbackLightEnvelope("DISTANCE_SUPPRESSED");
    }

    // Apply visibility overrides directly to target lighting profiles
    if (input.visibilityClass === "LIGHT_ONLY" && targetMode !== "GHOST") {
      targetMode = "POINT";
    }

    // Map specific layout configurations to their internal category tags
    if (targetMode === "DUAL_NAV") reason = "LIGHT_DUAL_NAV";
    else if (targetMode === "CLUSTER") reason = "LIGHT_CLUSTER";
    else if (targetMode === "POINT") reason = "LIGHT_POINT";
    else if (targetMode === "GHOST") reason = "LIGHT_GHOST";

    // 5. Calculate Color Parameters and Transparency Weights
    const baseAlphaCalculated = signature.baseAlpha * pulseVal * distMultiplier;
    const finalAlpha = Math.min(Math.max(baseAlphaCalculated + activeShimmer, MIN_LIGHT_ALPHA), MAX_NAV_ALPHA);

    // Set up standard color codes for structural navigation lines
    const navPortColor = `rgba(235, 64, 52, ${finalAlpha})`;       // Standard Port Red
    const navStarboardColor = `rgba(52, 235, 110, ${finalAlpha})`;  // Standard Starboard Green
    const navSternColor = `rgba(245, 245, 240, ${finalAlpha * 0.9})`; // Soft Stern White
    
    // Balance lighting warmth factors across amber and white color channels
    const warmthRatio = signature.farWarmth;
    const rGlow = Math.floor(245 * (1.0 - warmthRatio) + 255 * warmthRatio);
    const gGlow = Math.floor(245 * (1.0 - warmthRatio) + 180 * warmthRatio);
    const bGlow = Math.floor(250 * (1.0 - warmthRatio) + 60 * warmthRatio);
    const glowColor = `rgba(${rGlow}, ${gGlow}, ${bGlow}, ${finalAlpha * 0.4})`;

    // 6. Compute Final Edge Bloom Constraints
    const baseBloomRadius = 3.5 * signature.bloomScale * (1.5 - input.zoom / 15);
    const finalBloomRadius = Math.min(Math.max(baseBloomRadius, MIN_BLOOM_RADIUS_PX), MAX_BLOOM_RADIUS_PX);
    const finalBloomAlpha = Math.min(MAX_BLOOM_ALPHA * signature.baseAlpha * (1.0 - (input.fogAlpha ?? 0.0)), MAX_BLOOM_ALPHA);

    return {
      version: "1.0.0",
      visible: true,
      renderMode: targetMode,
      reasonCode: reason,
      alpha: finalAlpha,
      bloomAlpha: finalBloomAlpha,
      bloomRadiusPx: finalBloomRadius,
      navAlpha: (targetMode === "DUAL_NAV" || targetMode === "CLUSTER") ? finalAlpha : 0.0,
      farAlpha: (targetMode === "POINT" || targetMode === "GHOST") ? finalAlpha * 0.8 : 0.0,
      pulsePhase,
      pulseValue: pulseVal,
      shimmerAmount: activeShimmer,
      navPortColor,
      navStarboardColor,
      navSternColor,
      glowColor,
      allowNavPair: targetMode === "DUAL_NAV",
      allowMastLight: targetMode === "DUAL_NAV" || targetMode === "CLUSTER",
      allowFarGlint: targetMode === "POINT" || targetMode === "GHOST",
      allowBloom: targetMode !== "GHOST" && finalBloomAlpha > 0.01,
      allowWakeGlow: targetMode === "DUAL_NAV" && distanceBand === "HERO",
      allowReflectionHint: distanceBand === "HERO" || distanceBand === "NEAR"
    };
  }

  public getFallbackLightEnvelope(reasonCode: MaritimeLightReasonCode): MaritimeLightEnvelope {
    return {
      version: "1.0.0", visible: false, renderMode: "NONE", reasonCode,
      alpha: 0, bloomAlpha: 0, bloomRadiusPx: 0, navAlpha: 0, farAlpha: 0,
      pulsePhase: 0, pulseValue: 0, shimmerAmount: 0,
      navPortColor: "rgba(0,0,0,0)", navStarboardColor: "rgba(0,0,0,0)", navSternColor: "rgba(0,0,0,0)", glowColor: "rgba(0,0,0,0)",
      allowNavPair: false, allowMastLight: false, allowFarGlint: false, allowBloom: false, allowWakeGlow: false, allowReflectionHint: false
    };
  }

  public getConstants(): object {
    return { MIN_LIGHT_ALPHA, MAX_NAV_ALPHA, MAX_FAR_ALPHA, MAX_BLOOM_RADIUS_PX, MAX_BLOOM_ALPHA };
  }
}
