# 1. CORE ARCHITECTURAL DOCTRINE

The `MaritimeDistanceAtmosphere` layer calculates how live maritime assets step down in detail, soften, and blend into ambient water layers based on their distance from the focal viewpoint.

- **Distance is presentation truth. It is not simulation truth.**
- **Purely Interpretive Execution:** The module scales back visual attributes (such as opacities, trail lengths, glow values, and labeling rules) without touching entity logic or tracking flags.
- **2D owns truth. 2.5D owns presentation.**
- **Unidirectional Masking Bounds:** Distance-based weather masking can only step down or clamp alphas further. It cannot override or boost details that have already been restricted by upstream visibility rules.
- **Isolating Active Calculations:** To maximize runtime performance, depth evaluations run exactly once per asset per frame loop.

---

# 2. RUNTIME AUTHORITY BOUNDARIES

### 2.1 Explicit Domain Ownership
The module asserts exclusive governance over:
- Defining and checking boundary thresholds for the 5 canonical distance bands (`HERO`, `NEAR`, `MID`, `FAR`, `ATMOSPHERIC`).
- Generating the master `MaritimeDistanceEnvelope` structure used across drawing passes.
- Computing combined atmospheric masking values using regional fog, haze, and data-density variables.
- Enforcing label and hover-card display rules based on viewport distance.
- Providing structural level-of-detail (LOD) hints to active drawing engines.

### 2.2 Prohibited Modification Scopes
The module has **zero authority** over, and is completely blocked from mutating:
- Absolute location variables, headings, speeds, or identification keys in `AISRuntime`.
- Global weather variables or atmospheric state variables tracked by `VisibilityClassRuntime`.
- Master style configuration tables or color palettes managed within the `MaritimeStyleRegistry`.
- Active canvas viewport layouts, rendering layer assignments, or operational loops.

EXECUTABLE DISTANCE COMPRESSION ENGINETypeScriptexport class MaritimeDistanceAtmosphereModule {
  constructor() {}

  public resolveDistanceNorm(input: MaritimeDistanceInput): number {
    if (!input.viewportWidth || !input.viewportHeight) return 0.5; // Safe fallback standard
    const cx = input.viewportWidth * 0.5;
    const cy = input.viewportHeight * 0.5;
    const dx = input.screenX - cx;
    const dy = input.screenY - cy;
    const maxD = Math.sqrt(cx * cx + cy * cy);
    return Math.min(Math.max(Math.sqrt(dx * dx + dy * dy) / maxD, 0), 1);
  }

  public resolveDistanceBand(distanceNorm: number): MaritimeDistanceBand {
    if (distanceNorm <= HERO_RADIUS_NORM) return "HERO";
    if (distanceNorm <= NEAR_RADIUS_NORM) return "NEAR";
    if (distanceNorm <= MID_RADIUS_NORM) return "MID";
    if (distanceNorm <= FAR_RADIUS_NORM) return "FAR";
    return "ATMOSPHERIC";
  }

  public resolveDistanceEnvelope(input: MaritimeDistanceInput): MaritimeDistanceEnvelope {
    const dNorm = this.resolveDistanceNorm(input);
    const band = this.resolveDistanceBand(dNorm);

    // Compute weather-based suppression scales
    const fog = input.fogAlpha ?? 0.0;
    const haze = input.hazeAlpha ?? 0.0;
    const density = input.densityPressure ?? 0.0;
    
    const atmosphereNorm = Math.min(
      Math.max(
        fog * DEFAULT_FOG_WEIGHT + 
        haze * DEFAULT_HAZE_WEIGHT + 
        density * DEFAULT_DENSITY_WEIGHT, 
        0
      ), 
      1
    );

    // Pull initial alpha boundaries from the configuration matrix
    let vA = 1.0; let tA = 1.0; let wA = 1.0; let lA = 1.0; let lbA = 1.0; let hA = 1.0;
    let tDetail = 1.0; let wDetail = 1.0; let lBloom = 1.0;

    switch (band) {
      case "HERO":
        vA = 1.00; tA = 1.00; wA = 1.00; lA = 1.00; lbA = 1.00; hA = 1.00;
        tDetail = 1.0; wDetail = 1.0; lBloom = 1.0;
        break;
      case "NEAR":
        vA = 0.88; tA = 0.88; wA = 0.72; lA = 0.90; lbA = 0.80; hA = 0.75;
        tDetail = 1.0; wDetail = 0.8; lBloom = 1.1;
        break;
      case "MID":
        vA = 0.62; tA = 0.48; wA = 0.32; lA = 0.72; lbA = 0.20; hA = 0.00;
        tDetail = 0.5; wDetail = 0.4; lBloom = 1.3;
        break;
      case "FAR":
        vA = 0.32; tA = 0.18; wA = 0.08; lA = 0.55; lbA = 0.00; hA = 0.00;
        tDetail = 0.0; wDetail = 0.1; lBloom = 1.5;
        break;
      case "ATMOSPHERIC":
        vA = 0.10; tA = 0.00; wA = 0.00; lA = 0.35; lbA = 0.00; hA = 0.00;
        tDetail = 0.0; wDetail = 0.0; lBloom = 1.8;
        break;
    }

    // Apply linear degradation based on localized weather factors
    const weatherInverse = 1.0 - atmosphereNorm;
    vA *= weatherInverse;
    tA *= weatherInverse;
    wA *= weatherInverse;
    lbA *= weatherInverse * 0.5;
    hA *= weatherInverse * 0.2;

    // Build operational eligibility boolean metrics
    const envelope: MaritimeDistanceEnvelope = {
      version: "1.0.0",
      band,
      distanceNorm: dNorm,
      zoomNorm: Math.min(Math.max((input.zoom - 11) / 4, 0), 1),
      atmosphereNorm,
      vesselAlpha: Math.max(vA, MIN_ATMOSPHERIC_ALPHA),
      topologyAlpha: tA,
      wakeAlpha: wA,
      lightAlpha: Math.min(lA, MAX_FAR_LIGHT_ALPHA),
      labelAlpha: lbA,
      hoverAlpha: hA,
      topologyDetailScale: tDetail,
      wakeDetailScale: wDetail,
      lightBloomScale: lBloom,
      allowWake: wA > 0.02 && band !== "ATMOSPHERIC" && band !== "FAR",
      allowTopology: tA > 0.02 && band !== "ATMOSPHERIC",
      allowLabel: lbA > 0.05 && (band === "HERO" || band === "NEAR"),
      allowHover: hA > 0.05 && band === "HERO",
      allowNavLights: band === "HERO" || band === "NEAR" || band === "MID",
      allowFarLight: true,
      reason: `Distance classification verified as ${band}.`
    };

    // Apply parental overrides from VisibilityClassRuntime
    return this.applyVisibilityClassToEnvelope(envelope, input.visibilityClass);
  }

  public applyVisibilityClassToEnvelope(
    envelope: MaritimeDistanceEnvelope, 
    vClass: string | null
  ): MaritimeDistanceEnvelope {
    if (!vClass) return envelope;

    switch (vClass.toUpperCase().trim()) {
      case "ATMOSPHERIC_HIDDEN":
        return {
          ...envelope,
          vesselAlpha: 0, topologyAlpha: 0, wakeAlpha: 0, lightAlpha: 0, labelAlpha: 0, hoverAlpha: 0,
          allowWake: false, allowTopology: false, allowLabel: false, allowHover: false, allowNavLights: false, allowFarLight: false,
          reason: "Overridden by visibility rule: ATMOSPHERIC_HIDDEN."
        };
      case "LIGHT_ONLY":
        return {
          ...envelope,
          vesselAlpha: 0.1, topologyAlpha: 0, wakeAlpha: 0, labelAlpha: 0, hoverAlpha: 0,
          allowWake: false, allowTopology: false, allowLabel: false, allowHover: false, allowNavLights: false,
          reason: "Overridden by visibility rule: LIGHT_ONLY."
        };
      case "MARKER_ONLY":
        return {
          ...envelope,
          topologyAlpha: 0.1, wakeAlpha: 0, labelAlpha: 0, hoverAlpha: 0,
          allowWake: false, allowLabel: false, allowHover: false,
          reason: "Overridden by visibility rule: MARKER_ONLY."
        };
      default:
        return envelope;
    }
  }

  public getConstants(): object {
    return { HERO_RADIUS_NORM, NEAR_RADIUS_NORM, MID_RADIUS_NORM, FAR_RADIUS_NORM, DEFAULT_FOG_WEIGHT };
  }
}
