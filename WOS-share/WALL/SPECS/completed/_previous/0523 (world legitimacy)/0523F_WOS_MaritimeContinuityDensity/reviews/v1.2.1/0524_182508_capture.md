# 1. CORE ARCHITECTURAL DOCTRINE

- **WakeAuthority may preserve water memory.**

function emitWakeSegment(input: WakeAuthorityInput): WakeSegment | null {
  const calculatedClass = resolveWakeAuthorityClass(input.vesselClass);
  if (calculatedClass === "NONE" || input.speedKts < 2.0) {
    return null;
  }

  const generatedId = `wake::${input.vesselId}::${input.simulationTimeMs}`;
  const lastPos = lastPositionMap.get(input.vesselId);
  
  const startLatitude = lastPos ? lastPos.lat : input.lat;
  const startLongitude = lastPos ? lastPos.lng : input.lng;

  const lifetimeDuration = getTierLifetime(input.populationTier, input.provenance);

  const segment: WakeSegment = {
    wakeId: generatedId,
    vesselId: input.vesselId,
    provenance: input.provenance,
    wakeClass: calculatedClass,
    startLat: startLatitude,
    startLng: startLongitude,
    endLat: input.lat,
    endLng: input.lng,
    createdAtMs: input.simulationTimeMs,
    expiresAtMs: input.simulationTimeMs + lifetimeDuration,
    intensityRaw: Math.min(input.speedKts / 30.0, 1.00),
    parentEvicted: false
  };

  // Safe Ring-Buffer Write Optimization Logic
  let targetIndex = headIndex;
  let slotOccupied = wakeRingBuffer[targetIndex];

  if (slotOccupied && slotOccupied.provenance === "AIS_VESSEL" && segment.provenance === "SYNTHETIC_ECOLOGY") {
    // Provenance Protection rule: Synthetic tracks cannot overwrite active real-world data positions
    return null; 
  }

  wakeRingBuffer[targetIndex] = segment;
  headIndex = (targetIndex + 1) % MAX_WAKE_SEGMENTS;

  lastPositionMap.set(input.vesselId, { lat: input.lat, lng: input.lng });
  return segment;
}

function decayWakeSegments(simulationTimeMs: number): void {
  for (let i = 0; i < MAX_WAKE_SEGMENTS; i++) {
    const seg = wakeRingBuffer[i];
    if (seg && simulationTimeMs >= seg.expiresAtMs) {
      wakeRingBuffer[i] = null; // Purge expired tracking nodes from memory
    }
  }
}

function notifyAISSignalGap(vesselId: string): void {
  lastPositionMap.delete(vesselId); // Reset tracking anchors to preserve data gap rules
}

function notifyVesselEvicted(vesselId: string, simulationTimeMs: number): void {
  lastPositionMap.delete(vesselId);
  for (let i = 0; i < MAX_WAKE_SEGMENTS; i++) {
    const seg = wakeRingBuffer[i];
    if (seg && seg.vesselId === vesselId && !seg.parentEvicted) {
      seg.parentEvicted = true;
      const remaining = seg.expiresAtMs - simulationTimeMs;
      if (remaining > 0) {
        seg.expiresAtMs = simulationTimeMs + Math.floor(remaining / 4);
      }
    }
  }
}
