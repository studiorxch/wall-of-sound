# 1. CORE ARCHITECTURAL DOCTRINE

The `MaritimeContinuityDensity` module serves as a passive, deterministic observability layer inside the Wall of Sound (WOS) system architecture. It continuously gauges, aggregates, and exports occupancy weights, tracking burdens, and clutter markers across spatial cells without mutating real-world entity tracking engines.

- **ContinuityDensity may describe pressure.**
- **ContinuityDensity may never resolve pressure by mutating truth.**

### 1.1 Non-Mutation Boundaries
The component is fully read-only. High occupancy parameters or sector saturation flags can never authorize this system to cull entities, suppress active network packets, drop synthetic vessel profiles, accelerate trail degradation, or force scheduling shifts.

4.1 Deterministic Continuity Burden LoopA target vessel track is flagged as stale or low-confidence if it satisfies any tracking degradation conditions (signalConfidence < 0.40, continuityAlpha < 0.50, or matching unaligned navigation lifecycles).TypeScriptfunction calculateContinuityLoad(vessels: readonly MaritimeDensityVesselInput[]): number {
  if (vessels.length <= 0) return 0.00;

  let totalSignalBurden = 0.00;
  let totalAlphaBurden = 0.00;
  let staleCount = 0;

  for (let i = 0; i < vessels.length; i++) {
    const vessel = vessels[i];
    totalSignalBurden += (1.00 - vessel.signalConfidence);
    totalAlphaBurden += vessel.continuityAlpha;

    const isStale =
      vessel.signalConfidence < 0.40 ||
      vessel.continuityAlpha < 0.50 ||
      vessel.lifecycleState === "COASTING" ||
      vessel.lifecycleState === "DORMANT" ||
      vessel.lifecycleState === "STALE";

    if (isStale) {
      staleCount++;
    }
  }

  const avgSignalBurden = totalSignalBurden / vessels.length;
  const avgAlphaBurden = totalAlphaBurden / vessels.length;
  const staleRatio = staleCount / vessels.length;

  return clamp(
    (avgSignalBurden * 0.45) +
    (avgAlphaBurden * 0.35) +
    (staleRatio * 0.20),
    0.00,
    1.00
  );
}

COMPONENT FUNCTION SCHEMAS8.1 resolveSectorDensityTypeScriptfunction resolveSectorDensity(sector: MaritimeDensitySectorInput, simulationTimeMs: number): MaritimeDensitySectorResult {
  let hero = 0; let mid = 0; let bg = 0; let ghost = 0;
  let ais = 0;  let synth = 0;

  for (let i = 0; i < sector.vessels.length; i++) {
    const v = sector.vessels[i];
    if (v.provenance === "AIS_VESSEL") ais++; else synth++;
    
    if (v.populationTier === "HERO") hero++;
    else if (v.populationTier === "MID") mid++;
    else if (v.populationTier === "BACKGROUND") bg++;
    else if (v.populationTier === "GHOST") ghost++;

  const vDensity = resolveVesselDensityScore(sector.vessels.length, sector.budget.maxVessels);
  const wDensity = resolveWakeDensityScore(sector.wakes.length, sector.budget.maxWakeSegments);

8.2 resolveContinuityDensityTypeScriptfunction resolveContinuityDensity(input: MaritimeContinuityDensityInput): MaritimeContinuityDensityResult {
  const totalSectors = input.sectors.length;
  if (totalSectors === 0) {
    return {
      simulationTimeMs: input.simulationTimeMs,
      globalDensityScore: 0, globalContinuityLoad: 0, globalClutterPressure: 0,
      vesselDensityScore: 0, wakeDensityScore: 0, syntheticPressure: 0, aisPressure: 0,
      densityClass: "SPARSE", sectors: [], reasonCodes: ["DENSITY_NOMINAL"]
    };
  }

  let sectorResults: MaritimeDensitySectorResult[] = [];
  let sumSectorDensity = 0;
  let sumContinuityLoad = 0;
  let sumClutterPressure = 0;
  
  let totalVessels = 0;      let totalWakes = 0;
  let totalAis = 0;          let totalSynth = 0;
  let totalBg = 0;           let totalGhost = 0;

  for (let i = 0; i < totalSectors; i++) {
    const res = resolveSectorDensity(input.sectors[i], input.simulationTimeMs);
    sectorResults.push(res);

    sumSectorDensity += res.vesselDensityScore; // Linear baseline mapping
    sumContinuityLoad += res.continuityLoad;
    sumClutterPressure += res.clutterPressure;

    totalVessels += res.vesselCount;
    totalWakes += res.wakeCount;
    totalAis += res.aisCount;
    totalSynth += res.syntheticCount;
    totalBg += res.backgroundCount;
    totalGhost += res.ghostCount;
  }

  const avgContinuityLoad = sumContinuityLoad / totalSectors;
  const avgClutterPressure = sumClutterPressure / totalSectors;

  const globalVesselD = resolveVesselDensityScore(totalVessels, input.globalBudgetState.maxVessels);
  const globalWakeD = resolveWakeDensityScore(totalWakes, input.globalBudgetState.maxWakeSegments);
  const globalSynthP = resolveSyntheticPressure(totalSynth, input.globalBudgetState.maxSyntheticVessels);
  const globalAisP = resolveAISPressure(totalAis, input.globalBudgetState.maxVessels);

  const globalScore = clamp(
    (globalVesselD * 0.45) + (globalWakeD * 0.25) + (globalSynthP * 0.15) + (avgContinuityLoad * 0.15),
    0.00,
    1.00
  );

  let reasons: MaritimeContinuityDensityReason[] = [];
  if (globalAisP > 0.70) reasons.push("HIGH_AIS_PRESSURE");
  if (globalSynthP > 0.70) reasons.push("HIGH_SYNTHETIC_PRESSURE");
  if (globalWakeD > 0.70) reasons.push("HIGH_WAKE_PRESSURE");
  if (totalBg > totalVessels * 0.5) reasons.push("HIGH_BACKGROUND_PRESSURE");
  if (totalGhost > totalVessels * 0.3) reasons.push("HIGH_GHOST_PRESSURE");
  if (globalScore > 0.75) reasons.push("CONTINUITY_LOAD_HIGH");
  if (reasons.length === 0) reasons.push("DENSITY_NOMINAL");

  return {
    simulationTimeMs: input.simulationTimeMs,
    globalDensityScore: globalScore,
    globalContinuityLoad: avgContinuityLoad,
    globalClutterPressure: avgClutterPressure,
    vesselDensityScore: globalVesselD,
    wakeDensityScore: globalWakeD,
    syntheticPressure: globalSynthP,
    aisPressure: globalAisP,
    densityClass: resolveDensityClass(globalScore),
    sectors: sectorResults,
    reasonCodes: reasons
  };
}
