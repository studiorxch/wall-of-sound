// Immutable Traversal Scaling Ladders
const SPEED_LADDER = [0.25, 0.5, 1, 2, 5, 10, 20, 40, 80];
const ALTITUDE_ZOOM_MAPPING = [
  { zoom: 15, altitudeFt: 250,    label: "Ground / Approach" },
  { zoom: 14, altitudeFt: 1500,   label: "Low Flight" },
  { zoom: 13, altitudeFt: 5000,   label: "City / Helicopter" },
  { zoom: 12, altitudeFt: 12500,  label: "Regional Flight" },
  { zoom: 11, altitudeFt: 35000,  label: "Airline Cruise" }
];

const ZOOM_OUT_CEILING = 11; // Absolute ceiling limit for normal flight navigation vectors

export class TraversalControlDeck {
  constructor(initialConfig = {}) {
    this.state = {
      speedIndex: 2, // Default to 1x index
      zoomIndex: 2,  // Default to Zoom 13 (5000ft) index
      actorType: initialConfig.actorType || 'aircraft',
      povType: initialConfig.povType || 'forward',
      pitch: initialConfig.pitch || 45,
      bearing: initialConfig.bearing || 0,
      routeDistanceKm: initialConfig.routeDistanceKm || 0,
      startTimeReal: Date.now(),
      startTimeSim: Date.now()
    };
  }

  // --- Task 1: Stepped Speed Controls ---
  public getSpeedMultiplier() {
    return SPEED_LADDER[this.state.speedIndex];
  }

  public stepSpeedUp() {
    if (this.state.speedIndex < SPEED_LADDER.length - 1) {
      this.state.speedIndex++;
    }
    return this.getSpeedMultiplier();
  }

  public stepSpeedDown() {
    if (this.state.speedIndex > 0) {
      this.state.speedIndex--;
    }
    return this.getSpeedMultiplier();
  }

  // --- Task 2: Stepped Altitude Controls ---
  public getCurrentAltitudeMapping() {
    return ALTITUDE_ZOOM_MAPPING[this.state.zoomIndex];
  }

  public stepAltitudeUp() {
    // Stepping altitude up means zooming OUT (decreasing zoom level index towards ceiling)
    if (this.state.zoomIndex < ALTITUDE_ZOOM_MAPPING.length - 1) {
      const targetMap = ALTITUDE_ZOOM_MAPPING[this.state.zoomIndex + 1];
      if (targetMap.zoom >= ZOOM_OUT_CEILING) {
        this.state.zoomIndex++;
      }
    }
    return this.getCurrentAltitudeMapping();
  }

  public stepAltitudeDown() {
    // Stepping altitude down means zooming IN (increasing zoom level index)
    if (this.state.zoomIndex > 0) {
      this.state.zoomIndex--;
    }
    return this.getCurrentAltitudeMapping();
  }

  // --- Real vs Sim Temporal Accumulators ---
  public calculateElapsedTime(currentTimestamp) {
    const deltaRealMs = currentTimestamp - this.state.startTimeReal;
    const deltaSimMs = deltaRealMs * this.getSpeedMultiplier();

    return {
      realMs: deltaRealMs,
      simMs: deltaSimMs
    };
  }

  /**

 * @fileoverview WOS Traversal HUD Telemetry Matrix Engine

export class TraversalHUD {
  constructor(controlDeckInstance) {
    this.deck = controlDeckInstance;
  }

  private formatTime(totalMs) {
    const totalSeconds = Math.floor(totalMs / 1000);
    const minutes = Math.floor(totalSeconds / 60);
    const seconds = totalSeconds % 60;
    return `${minutes}m ${seconds.toString().padStart(2, '0')}s`;
  }

  // --- Task 3: Render Telemetry Projection Output ---
  public compileHUDTelemetry(currentTimestamp) {
    const elapsed = this.deck.calculateElapsedTime(currentTimestamp);
    const altMap = this.deck.getCurrentAltitudeMapping();

    return {
      // Corrected Temporal Metrics
      REAL_elapsed: this.formatTime(elapsed.realMs),
      SIM_elapsed: this.formatTime(elapsed.simMs),
      speedMultiplier: `${this.deck.getSpeedMultiplier()}x`,

      // Absolute Presentation Layout Constraints
      altitudeFt: `${altMap.altitudeFt} ft`,
      zoom: altMap.zoom,
      pitch: `${this.deck.state.pitch}｡`,
      bearing: `${this.deck.state.bearing}｡`,

      // Target Taxonomy Identification Keys
      actorType: this.deck.state.actorType,
      povType: this.deck.state.povType,

      // Base Traversal Metadata Stays Intact
      statusStrings: {
        speedControlDisplay: `Speed [-] [${this.deck.getSpeedMultiplier()}x] [+]`,
        altitudeControlDisplay: `Altitude [-] [${altMap.altitudeFt} ft / Z${altMap.zoom}] [+]`
      }
    };
  }
}
